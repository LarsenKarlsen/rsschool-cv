def even_odd(arr: list[int]) -> float:
    """
    The function returns the ratio of the sum of even array elements to the sum of odd ones.
    Example:
    even_odd([1, 2, 3, 4, 5]) == 0.8889
    """
    sums = [0, 0]
    for num in arr:
        if num % 2 == 0:
            sums[0] += num
        else:
            sums[1] += num
    return sums[0] / sums[1] if sums[1] > 0 else 0
